4.1 Defining Functions

Functions are first-class objects in Python. Unlike C++/Java, they are defined at runtime and can be assigned to variables, stored in data structures, and passed as arguments.

def greet(name: str) -> str:
    """Return a greeting string."""
    return f"Hello, {name}!"

# Functions are objects
say_hello = greet
print(say_hello("World"))  # Hello, World!
Python functions always return a value. If no return is given, the function returns None.

4.2 Parameter Types

# Positional and keyword arguments
def connect(host, port=3306, timeout=30):
    print(f"Connecting to {host}:{port} (timeout={timeout}s)")

connect("db.server.com")                   # positional
connect("db.server.com", port=5432)         # keyword
connect("db.server.com", timeout=10)        # keyword

# Variable positional (*args)
def total(*values):
    return sum(values)
print(total(1, 2, 3, 4))   # 10

# Variable keyword (**kwargs)
def build_config(**kwargs):
    return {k: v for k, v in kwargs.items()}
print(build_config(debug=True, port=8080))

Keyword-Only and Positional-Only Parameters

# Keyword-only (after *)
def safe_div(a, b, *, round_digits=2):
    return round(a / b, round_digits)

# Positional-only (before /)  (Python 3.8+)
def power(base, exp, /):
    return base ** exp
Mutable Default Argument Trap: Never use a mutable object (list, dict) as a default parameter value. Use None and create the object inside the function body.
# WRONG - the list is shared across all calls
def append_to(item, target=[]):
    target.append(item)
    return target

# CORRECT
def append_to(item, target=None):
    if target is None:
        target = []
    target.append(item)
    return target

4.3 Scope and the LEGB Rule

Python resolves names using the LEGB order: Local → Enclosing → Global → Built-in.

x = "global"

def outer():
    x = "enclosing"
    def inner():
        x = "local"
        print(x)          # local
    inner()
    print(x)              # enclosing

outer()
print(x)                  # global

global and nonlocal

counter = 0
def increment():
    global counter
    counter += 1

def make_counter():
    count = 0
    def inc():
        nonlocal count
        count += 1
        return count
    return inc

4.4 Lambda Expressions

Lambda creates small anonymous functions. They are limited to a single expression.

# Sort by second element
pairs = [(1, "b"), (3, "a"), (2, "c")]
pairs.sort(key=lambda p: p[1])
print(pairs)  # [(3, 'a'), (1, 'b'), (2, 'c')]

# Equivalent to Java lambda:  (p) -> p[1]
# Equivalent to C++ lambda:   [](auto p){ return p.second; }

4.5 Closures and Higher-Order Functions

def make_multiplier(factor):
    """Return a function that multiplies by factor."""
    def multiplier(x):
        return x * factor
    return multiplier

double = make_multiplier(2)
triple = make_multiplier(3)
print(double(5))   # 10
print(triple(5))   # 15

Built-in Higher-Order Functions

nums = [1, 2, 3, 4, 5]

squared = list(map(lambda x: x**2, nums))       # [1, 4, 9, 16, 25]
evens = list(filter(lambda x: x % 2 == 0, nums)) # [2, 4]

from functools import reduce
total = reduce(lambda a, b: a + b, nums)          # 15

4.6 Type Hints and Docstrings

def binary_search(arr: list[int], target: int) -> int:
    """Return the index of target in sorted arr, or -1 if not found.

    Args:
        arr: A sorted list of integers.
        target: The integer to search for.

    Returns:
        The index of target, or -1.
    """
    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1
    return -1
Type hints are optional and not enforced at runtime. Use tools like mypy for static type checking, similar to TypeScript's relationship with JavaScript.
