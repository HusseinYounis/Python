5.1 Lists

A list is a mutable, ordered sequence — similar to std::vector<T> in C++ or ArrayList<T> in Java, but heterogeneous (elements can differ in type).

Creating Lists

nums = [10, 20, 30, 40, 50]
empty = []
mixed = [1, "hello", 3.14, True]     # heterogeneous
repeated = [0] * 5                    # [0, 0, 0, 0, 0]
from_range = list(range(1, 6))       # [1, 2, 3, 4, 5]
nested = [[1, 2], [3, 4], [5, 6]]   # 2D list (list of lists)

Indexing and Slicing

nums = [10, 20, 30, 40, 50]

# Indexing (0-based, same as C++/Java)
print(nums[0])      # 10
print(nums[-1])     # 50 (last element)
print(nums[-2])     # 40 (second to last)

# Slicing [start:stop:step]
print(nums[1:4])    # [20, 30, 40]
print(nums[:3])     # [10, 20, 30]     (first 3)
print(nums[2:])     # [30, 40, 50]     (from index 2)
print(nums[::2])    # [10, 30, 50]     (every other)
print(nums[::-1])   # [50, 40, 30, 20, 10] (reversed)

# Slice assignment (replace a section)
nums[1:3] = [200, 300, 350]
print(nums)          # [10, 200, 300, 350, 40, 50]

Common List Methods

nums = [10, 20, 30, 40, 50]

nums.append(60)           # Add to end - O(1)
nums.insert(0, 5)         # Insert at index - O(n)
nums.extend([70, 80])     # Append iterable - O(k)
val = nums.pop()          # Remove & return last - O(1)
val = nums.pop(2)         # Remove & return at index - O(n)
nums.remove(30)           # Remove first occurrence - O(n)

# Searching
print(nums.index(40))     # Index of first occurrence
print(nums.count(10))     # Number of occurrences

# Reordering
nums.reverse()            # In-place reverse
nums.sort()               # In-place sort - O(n log n)
sorted_copy = sorted(nums) # Return new sorted list (original unchanged)
nums.clear()              # Remove all elements

Sorting with Custom Keys

# Sort strings by length
words = ["banana", "pie", "apple", "fig"]
words.sort(key=len)
print(words)  # ['pie', 'fig', 'apple', 'banana']

# Sort by absolute value
nums = [-5, 3, -1, 4, -2]
print(sorted(nums, key=abs))  # [-1, -2, 3, 4, -5]

# Sort list of tuples by second element
students = [("Alice", 3.9), ("Bob", 3.5), ("Carol", 3.7)]
students.sort(key=lambda s: s[1], reverse=True)
print(students)  # [('Alice', 3.9), ('Carol', 3.7), ('Bob', 3.5)]

# Multi-key sort with operator.itemgetter
from operator import itemgetter
data = [("Math", 90), ("CS", 95), ("Math", 85), ("CS", 90)]
data.sort(key=itemgetter(0, 1))  # Sort by subject, then score
print(data)
# [('CS', 90), ('CS', 95), ('Math', 85), ('Math', 90)]

Unpacking and the * Operator

# Extended unpacking
first, *middle, last = [1, 2, 3, 4, 5]
print(first)    # 1
print(middle)   # [2, 3, 4]
print(last)     # 5

# Merge lists with *
a = [1, 2, 3]
b = [4, 5, 6]
merged = [*a, *b]         # [1, 2, 3, 4, 5, 6]
print(merged)

Shallow vs Deep Copy

import copy

a = [[1, 2], [3, 4]]

# Assignment is NOT a copy - same object
b = a
b.append([5, 6])
print(a)  # [[1, 2], [3, 4], [5, 6]] - a is also modified!

# Shallow copy - new outer list, but shared inner lists
c = a.copy()          # or: c = a[:] or c = list(a)
a[0][0] = 99
print(c[0][0])        # 99 - inner list is shared

# Deep copy - fully independent
d = copy.deepcopy(a)
a[0][0] = 0
print(d[0][0])        # 99 - independent copy

Common Pitfall: Mutable Default Arguments

# WRONG - the default list is shared across calls
def add_item_bad(item, lst=[]):
    lst.append(item)
    return lst

print(add_item_bad(1))   # [1]
print(add_item_bad(2))   # [1, 2] ← unexpected!

# CORRECT - use None as default
def add_item(item, lst=None):
    if lst is None:
        lst = []
    lst.append(item)
    return lst

print(add_item(1))       # [1]
print(add_item(2))       # [2] ← correct

5.2 Tuples

Tuples are immutable sequences. Once created, elements cannot be changed. They are hashable (if all elements are hashable) and can be used as dictionary keys or set elements.

Creating Tuples

point = (3.0, 4.0)
single = (42,)              # Single-element tuple NEEDS the comma
not_a_tuple = (42)          # This is just int 42!
empty = ()
from_list = tuple([1, 2, 3])

# Parentheses are optional in many contexts
rgb = 255, 128, 0           # Also a tuple

Tuple Unpacking

point = (3.0, 4.0)
x, y = point              # Unpack into variables

# Swap without temp variable
a, b = 1, 2
a, b = b, a                # a=2, b=1

# Unpack in loops
pairs = [(1, "a"), (2, "b"), (3, "c")]
for num, letter in pairs:
    print(f"{num} -> {letter}")

# Extended unpacking
first, *rest = (1, 2, 3, 4, 5)
print(first)  # 1
print(rest)   # [2, 3, 4, 5] - note: rest is a list

Tuple Methods and Operations

t = (1, 2, 3, 2, 1)
print(t.count(2))     # 2
print(t.index(3))     # 2 (first occurrence)
print(len(t))         # 5
print(t + (4, 5))     # (1, 2, 3, 2, 1, 4, 5) - concatenation
print(t * 2)          # (1, 2, 3, 2, 1, 1, 2, 3, 2, 1)

# Tuples as dictionary keys (lists CANNOT be)
locations = {}
locations[(40.7, -74.0)] = "New York"
locations[(51.5, -0.1)] = "London"

Named Tuples

from collections import namedtuple

Student = namedtuple("Student", ["name", "gpa", "major"])
s = Student("Alice", 3.9, "CE")
print(s.name, s.gpa)             # Alice 3.9
print(s[0])                       # Alice (index access also works)

# Convert to dict
print(s._asdict())  # {'name': 'Alice', 'gpa': 3.9, 'major': 'CE'}

# Create modified copy (tuples are immutable)
s2 = s._replace(gpa=4.0)
print(s2)  # Student(name='Alice', gpa=4.0, major='CE')

# typing.NamedTuple (modern alternative)
from typing import NamedTuple

class Point(NamedTuple):
    x: float
    y: float
    label: str = "origin"   # default value

p = Point(3.0, 4.0)
print(p.label)               # "origin"

5.3 Dictionaries

Dictionaries are hash maps — equivalent to std::unordered_map in C++ or HashMap in Java. Average O(1) lookup, insertion, and deletion. Since Python 3.7, dictionaries maintain insertion order.

Creating Dictionaries

# Literal syntax
student = {
    "name": "Alice",
    "gpa": 3.9,
    "courses": ["CS101", "MATH201"]
}

# From sequence of pairs
pairs = [("a", 1), ("b", 2), ("c", 3)]
d = dict(pairs)

# Using keyword arguments
config = dict(host="localhost", port=8080, debug=True)

# dict.fromkeys - same value for all keys
keys = ["x", "y", "z"]
zeroed = dict.fromkeys(keys, 0)   # {'x': 0, 'y': 0, 'z': 0}

Accessing and Modifying

student = {"name": "Alice", "gpa": 3.9}

# Access
print(student["name"])              # Alice
print(student.get("age", "N/A"))    # N/A (safe - no KeyError)

# Mutation
student["age"] = 21                 # Add new key
student["gpa"] = 4.0                # Update existing
del student["age"]                  # Remove key

# pop - remove and return value
gpa = student.pop("gpa")            # 4.0
missing = student.pop("foo", None)  # None (no KeyError)

# setdefault - get value, or set it if missing
student.setdefault("year", 3)       # Sets year=3, returns 3
student.setdefault("year", 4)       # year already exists, returns 3

Iteration Patterns

scores = {"Alice": 95, "Bob": 87, "Carol": 92}

# Iterate keys (default)
for name in scores:
    print(name)

# Iterate values
for score in scores.values():
    print(score)

# Iterate key-value pairs
for name, score in scores.items():
    print(f"{name}: {score}")

# Check membership (checks KEYS, not values)
print("Alice" in scores)    # True
print(95 in scores)          # False (95 is a value, not a key)
print(95 in scores.values()) # True

Merging Dictionaries

defaults = {"color": "blue", "size": 10, "visible": True}
overrides = {"size": 20, "border": True}

# Method 1: update() - modifies in place
config = defaults.copy()
config.update(overrides)
print(config)  # {'color': 'blue', 'size': 20, 'visible': True, 'border': True}

# Method 2: {**d1, **d2} unpacking
config = {**defaults, **overrides}

# Method 3: | merge operator (Python 3.9+)
config = defaults | overrides          # New dict
defaults |= overrides                  # In-place update

defaultdict and Counter

from collections import defaultdict, Counter

# defaultdict auto-creates missing keys with a factory
word_count = defaultdict(int)     # missing keys default to 0
for word in "the cat sat on the mat".split():
    word_count[word] += 1
print(dict(word_count))  # {'the': 2, 'cat': 1, 'sat': 1, ...}

# Group items with defaultdict(list)
students = [("CS", "Alice"), ("Math", "Bob"), ("CS", "Carol"), ("Math", "Dave")]
by_dept = defaultdict(list)
for dept, name in students:
    by_dept[dept].append(name)
print(dict(by_dept))  # {'CS': ['Alice', 'Carol'], 'Math': ['Bob', 'Dave']}

# Counter - specialized frequency counter
freq = Counter("abracadabra")
print(freq.most_common(3))  # [('a', 5), ('b', 2), ('r', 2)]

# Counter arithmetic
c1 = Counter(a=3, b=1)
c2 = Counter(a=1, b=2)
print(c1 + c2)   # Counter({'a': 4, 'b': 3})
print(c1 - c2)   # Counter({'a': 2})  - drops zero/negative

OrderedDict and ChainMap

from collections import OrderedDict, ChainMap

# OrderedDict - preserves order AND supports move_to_end / popitem(last=...)
od = OrderedDict([("a", 1), ("b", 2), ("c", 3)])
od.move_to_end("a")            # Move 'a' to end
print(list(od))                 # ['b', 'c', 'a']
od.move_to_end("c", last=False) # Move 'c' to beginning
print(list(od))                 # ['c', 'b', 'a']

# ChainMap - search multiple dicts as one
cli_args = {"debug": True}
env_vars = {"debug": False, "verbose": True}
defaults = {"debug": False, "verbose": False, "log": "info"}

config = ChainMap(cli_args, env_vars, defaults)
print(config["debug"])     # True  (from cli_args - first match wins)
print(config["verbose"])   # True  (from env_vars)
print(config["log"])       # "info" (from defaults)

5.4 Sets

Sets are unordered collections of unique, hashable elements — similar to std::unordered_set in C++ or HashSet in Java.

Creating Sets

a = {1, 2, 3, 4, 5}
from_list = set([1, 2, 2, 3, 3])   # {1, 2, 3} - duplicates removed
empty = set()                        # NOT {} - that creates an empty dict!
chars = set("mississippi")           # {'m', 'i', 's', 'p'}

Set Operations

a = {1, 2, 3, 4, 5}
b = {4, 5, 6, 7, 8}

# Operators and equivalent methods
print(a | b)    # Union:        {1, 2, 3, 4, 5, 6, 7, 8}
print(a.union(b))

print(a & b)    # Intersection: {4, 5}
print(a.intersection(b))

print(a - b)    # Difference:   {1, 2, 3}
print(a.difference(b))

print(a ^ b)    # Symmetric:    {1, 2, 3, 6, 7, 8}
print(a.symmetric_difference(b))

# Subset and superset
small = {1, 2}
print(small <= a)          # True (subset)
print(a >= small)          # True (superset)
print(small < a)           # True (proper subset)

Modifying Sets

s = {1, 2, 3}

s.add(4)                   # Add single element
s.update([5, 6, 7])        # Add multiple elements
s.discard(10)              # Remove if present (NO error if missing)
s.remove(3)                # Remove (KeyError if missing!)
val = s.pop()              # Remove and return arbitrary element
s.clear()                  # Remove all

frozenset — Immutable Sets

# frozenset is hashable, so it can be used as a dict key or in another set
fs = frozenset([1, 2, 3])

# Set of sets (must use frozenset)
groups = {frozenset({1, 2}), frozenset({3, 4})}

# frozenset supports all read operations
print(fs | {4, 5})   # frozenset({1, 2, 3, 4, 5})
print(3 in fs)        # True

# Use as dictionary key
cache = {}
cache[frozenset({"a", "b"})] = "result_ab"

Practical Examples

# Remove duplicates while preserving order
def unique_ordered(seq):
    seen = set()
    result = []
    for item in seq:
        if item not in seen:
            seen.add(item)
            result.append(item)
    return result

print(unique_ordered([3, 1, 4, 1, 5, 9, 2, 6, 5, 3]))
# [3, 1, 4, 5, 9, 2, 6]

# Find common elements across multiple lists
lists = [[1, 2, 3, 4], [2, 3, 4, 5], [3, 4, 5, 6]]
common = set(lists[0])
for lst in lists[1:]:
    common &= set(lst)
print(common)   # {3, 4}

5.5 Stacks and Queues

Python has no built-in stack or queue class, but lists and collections.deque serve these roles efficiently.

Stack (LIFO) Using a List

# list.append() and list.pop() give O(1) stack behavior
stack = []
stack.append("a")   # push
stack.append("b")
stack.append("c")
print(stack)         # ['a', 'b', 'c']

top = stack.pop()    # pop
print(top)           # 'c'
print(stack)         # ['a', 'b']

# Peek without removing
print(stack[-1])     # 'b'

Practical Stack Example: Balanced Brackets

def is_balanced(expr):
    """Check if brackets are balanced."""
    matching = {')': '(', ']': '[', '}': '{'}
    stack = []
    for ch in expr:
        if ch in '([{':
            stack.append(ch)
        elif ch in ')]}':
            if not stack or stack[-1] != matching[ch]:
                return False
            stack.pop()
    return len(stack) == 0

print(is_balanced("{[()]}"))    # True
print(is_balanced("{[(])}"))    # False
print(is_balanced("((())"))     # False

Queue (FIFO) Using collections.deque

Using list.pop(0) for a queue is O(n). Use deque for O(1) operations on both ends.

from collections import deque

# Queue: append to right, pop from left
queue = deque()
queue.append("task1")     # enqueue
queue.append("task2")
queue.append("task3")
print(queue)              # deque(['task1', 'task2', 'task3'])

first = queue.popleft()   # dequeue - O(1)
print(first)              # 'task1'
print(queue)              # deque(['task2', 'task3'])

deque as a Double-Ended Queue

from collections import deque

d = deque([1, 2, 3, 4, 5])

d.appendleft(0)           # Add to left   - O(1)
d.append(6)               # Add to right  - O(1)
d.popleft()               # Remove left   - O(1)
d.pop()                   # Remove right  - O(1)

# Rotate elements
d.rotate(2)               # Rotate right by 2
print(d)                  # deque([4, 5, 1, 2, 3])
d.rotate(-2)              # Rotate left by 2
print(d)                  # deque([1, 2, 3, 4, 5])

# Bounded deque (keeps last N items)
recent = deque(maxlen=3)
for i in range(5):
    recent.append(i)
print(recent)             # deque([2, 3, 4], maxlen=3)

Priority Queue with heapq

import heapq

# heapq implements a min-heap on a regular list
tasks = []
heapq.heappush(tasks, (3, "low priority"))
heapq.heappush(tasks, (1, "high priority"))
heapq.heappush(tasks, (2, "medium priority"))

while tasks:
    priority, name = heapq.heappop(tasks)
    print(f"[{priority}] {name}")
# [1] high priority
# [2] medium priority
# [3] low priority

# Find N smallest / largest
nums = [45, 12, 67, 3, 89, 23, 56]
print(heapq.nsmallest(3, nums))  # [3, 12, 23]
print(heapq.nlargest(3, nums))   # [89, 67, 56]

5.6 Nested Data Structures

Real-world data often combines lists, dicts, and tuples in nested structures — similar to JSON documents or database records.

Working with Nested Structures

# A university course catalog
catalog = {
    "CS101": {
        "title": "Intro to CS",
        "credits": 3,
        "students": [
            {"name": "Alice", "grade": "A"},
            {"name": "Bob", "grade": "B+"},
        ]
    },
    "MATH201": {
        "title": "Linear Algebra",
        "credits": 4,
        "students": [
            {"name": "Carol", "grade": "A-"},
        ]
    }
}

# Access nested data
print(catalog["CS101"]["title"])              # Intro to CS
print(catalog["CS101"]["students"][0]["name"]) # Alice

# Safely access deeply nested data
def safe_get(d, *keys, default=None):
    """Navigate nested dicts/lists without KeyError/IndexError."""
    for key in keys:
        try:
            d = d[key]
        except (KeyError, IndexError, TypeError):
            return default
    return d

print(safe_get(catalog, "CS101", "students", 0, "grade"))  # A
print(safe_get(catalog, "CS999", "title", default="N/A"))   # N/A

Building Nested Structures

# Build an adjacency list (graph)
edges = [("A", "B"), ("A", "C"), ("B", "C"), ("C", "D")]
graph = defaultdict(list)
for src, dst in edges:
    graph[src].append(dst)
    graph[dst].append(src)  # undirected

print(dict(graph))
# {'A': ['B', 'C'], 'B': ['A', 'C'], 'C': ['A', 'B', 'D'], 'D': ['C']}

# Invert a dictionary (swap keys and values)
grades = {"Alice": "A", "Bob": "B", "Carol": "A", "Dave": "B"}
by_grade = defaultdict(list)
for student, grade in grades.items():
    by_grade[grade].append(student)
print(dict(by_grade))  # {'A': ['Alice', 'Carol'], 'B': ['Bob', 'Dave']}
