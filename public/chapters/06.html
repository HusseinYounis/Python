6.1 Defining Classes

In Python, classes are defined with the class keyword. The constructor is __init__ (unlike C++ constructors or Java constructors that share the class name). Every instance method takes self as its first parameter (equivalent to this in C++/Java, but explicit).

class Vector2D:
    """A 2D vector with x and y components."""

    def __init__(self, x: float, y: float):
        self.x = x
        self.y = y

    def magnitude(self) -> float:
        return (self.x ** 2 + self.y ** 2) ** 0.5

    def __repr__(self) -> str:
        return f"Vector2D({self.x}, {self.y})"

v = Vector2D(3, 4)
print(v.magnitude())  # 5.0
print(v)              # Vector2D(3, 4)
Python has no new keyword. Calling Vector2D(3, 4) automatically creates the object and calls __init__.

6.2 Inheritance

class Shape:
    def area(self) -> float:
        raise NotImplementedError

class Circle(Shape):
    def __init__(self, radius: float):
        self.radius = radius

    def area(self) -> float:
        import math
        return math.pi * self.radius ** 2

class Rectangle(Shape):
    def __init__(self, width: float, height: float):
        self.width = width
        self.height = height

    def area(self) -> float:
        return self.width * self.height

# Polymorphism
shapes: list[Shape] = [Circle(5), Rectangle(3, 4)]
for s in shapes:
    print(f"{type(s).__name__}: area = {s.area():.2f}")

Multiple Inheritance and MRO

class A:
    def greet(self): return "A"

class B(A):
    def greet(self): return "B"

class C(A):
    def greet(self): return "C"

class D(B, C):
    pass

d = D()
print(d.greet())          # B (follows MRO)
print(D.__mro__)          # D -> B -> C -> A -> object
Python uses C3 Linearization for method resolution order (MRO). Java does not support multiple class inheritance (only interface inheritance). C++ supports it but with the diamond problem complexity.

6.3 Encapsulation and Properties

Python uses a convention-based approach instead of access modifiers (public, private, protected). A single underscore _attr means "protected by convention"; a double underscore __attr triggers name mangling.

class Temperature:
    def __init__(self, celsius: float):
        self._celsius = celsius   # "protected" by convention

    @property
    def celsius(self) -> float:
        return self._celsius

    @celsius.setter
    def celsius(self, value: float):
        if value < -273.15:
            raise ValueError("Below absolute zero")
        self._celsius = value

    @property
    def fahrenheit(self) -> float:
        return self._celsius * 9/5 + 32

t = Temperature(100)
print(t.fahrenheit)     # 212.0
t.celsius = -300        # ValueError!

6.4 Dunder Methods (Operator Overloading)

class Vector2D:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __add__(self, other):
        return Vector2D(self.x + other.x, self.y + other.y)

    def __eq__(self, other):
        return self.x == other.x and self.y == other.y

    def __len__(self):
        return 2

    def __getitem__(self, index):
        return (self.x, self.y)[index]

    def __repr__(self):
        return f"Vector2D({self.x}, {self.y})"

a = Vector2D(1, 2)
b = Vector2D(3, 4)
print(a + b)        # Vector2D(4, 6)
print(a == b)       # False
print(a[0])         # 1

6.5 Class Methods and Static Methods

class Date:
    def __init__(self, year, month, day):
        self.year = year
        self.month = month
        self.day = day

    @classmethod
    def from_string(cls, date_str: str):
        """Alternative constructor (factory method)."""
        y, m, d = map(int, date_str.split("-"))
        return cls(y, m, d)

    @staticmethod
    def is_valid(year, month, day) -> bool:
        return 1 <= month <= 12 and 1 <= day <= 31

d = Date.from_string("2024-03-15")
print(Date.is_valid(2024, 13, 1))  # False
