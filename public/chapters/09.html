9.1 Reading and Writing Text Files

Python file I/O is simpler than C++'s fstream or Java's BufferedReader. The with statement guarantees the file is closed properly.

# Writing
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("Line 1\n")
    f.write("Line 2\n")

# Reading entire file
with open("output.txt", "r", encoding="utf-8") as f:
    content = f.read()

# Reading line by line (memory efficient)
with open("output.txt") as f:
    for line in f:
        print(line.strip())

# Reading all lines into a list
with open("output.txt") as f:
    lines = f.readlines()

9.2 CSV Files

import csv

# Writing CSV
with open("students.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(["Name", "GPA", "Major"])
    writer.writerow(["Alice", 3.9, "CE"])
    writer.writerow(["Bob", 3.5, "CS"])

# Reading CSV
with open("students.csv", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['Name']}: {row['GPA']}")

9.3 Binary Files and struct

For engineers working with embedded systems or network protocols, binary I/O is essential. Python's struct module packs/unpacks C-compatible binary data.

import struct

# Pack data: unsigned short, float, 10-char string
header = struct.pack("<Hf10s", 42, 3.14, b"HELLO")

# Write binary
with open("data.bin", "wb") as f:
    f.write(header)

# Read and unpack
with open("data.bin", "rb") as f:
    data = f.read()
    id_val, pi_val, name = struct.unpack("<Hf10s", data)
    print(id_val, pi_val, name.decode().strip('\x00'))

9.4 Pickle Serialization

pickle serializes any Python object to a byte stream. It is Python-specific (not interoperable with Java/C++).

import pickle

data = {"model": "v2", "weights": [0.1, 0.5, 0.3], "accuracy": 0.95}

# Serialize
with open("model.pkl", "wb") as f:
    pickle.dump(data, f)

# Deserialize
with open("model.pkl", "rb") as f:
    loaded = pickle.load(f)
print(loaded["accuracy"])  # 0.95
Security Warning: Never unpickle data from untrusted sources. Malicious pickle files can execute arbitrary code. Use JSON or protocol buffers for untrusted data exchange.

9.5 Handling Large Files

# Process line by line (doesn't load entire file into memory)
def count_words(filepath):
    total = 0
    with open(filepath, encoding="utf-8") as f:
        for line in f:
            total += len(line.split())
    return total

# Read in chunks
def read_chunks(filepath, chunk_size=8192):
    with open(filepath, "rb") as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            yield chunk
