Testing and Debugging

Unit testing with pytest, assertions, mocking, and debugging techniques.

Learning Outcomes

  • ●Write unit tests with pytest and use assertions effectively.

11.1 pytest Basics

pytest is the de facto testing framework for Python. Unlike Java's JUnit, test functions don't need to be inside a class, and plain assert statements are used instead of special assertion methods.

# test_math_utils.py
def add(a, b):
    return a + b

def test_add_integers():
    assert add(2, 3) == 5

def test_add_floats():
    assert abs(add(0.1, 0.2) - 0.3) < 1e-9

def test_add_strings():
    assert add("hello ", "world") == "hello world"
# Run tests
# $ pytest test_math_utils.py -v
# test_math_utils.py::test_add_integers PASSED
# test_math_utils.py::test_add_floats PASSED
# test_math_utils.py::test_add_strings PASSED

11.2 Fixtures

Fixtures provide reusable test setup — similar to @Before/@After in JUnit or SetUp/TearDown in Google Test.

import pytest

@pytest.fixture
def sample_data():
    """Provide test data."""
    return {"users": ["Alice", "Bob"], "count": 2}

@pytest.fixture
def temp_file(tmp_path):
    """Create a temporary test file."""
    f = tmp_path / "test.txt"
    f.write_text("hello\nworld\n")
    return f

def test_user_count(sample_data):
    assert sample_data["count"] == len(sample_data["users"])

def test_file_lines(temp_file):
    lines = temp_file.read_text().strip().split("\n")
    assert len(lines) == 2

11.3 Parametrized Tests

@pytest.mark.parametrize("input_val,expected", [
    (1, 1),
    (2, 4),
    (3, 9),
    (0, 0),
    (-2, 4),
])
def test_square(input_val, expected):
    assert input_val ** 2 == expected

11.4 Mocking

from unittest.mock import patch, MagicMock

def fetch_data(url):
    import requests
    response = requests.get(url)
    return response.json()

@patch("requests.get")
def test_fetch_data(mock_get):
    # Configure mock
    mock_response = MagicMock()
    mock_response.json.return_value = {"status": "ok"}
    mock_get.return_value = mock_response

    # Test
    result = fetch_data("https://api.example.com/data")
    assert result == {"status": "ok"}
    mock_get.assert_called_once_with("https://api.example.com/data")

11.5 Debugging

# Built-in debugger (Python 3.7+)
def buggy_function(data):
    result = []
    for item in data:
        breakpoint()  # Drops into pdb here
        result.append(item * 2)
    return result

# PDB commands:
# n (next), s (step into), c (continue)
# p variable (print), l (list code), q (quit)

Logging vs Print

import logging

logging.basicConfig(level=logging.DEBUG)
logger = logging.getLogger(__name__)

def process(data):
    logger.debug(f"Processing {len(data)} items")
    logger.info("Starting computation")
    logger.warning("Memory usage high")
    logger.error("Failed to process item")
Use logging instead of print() for debugging in production code. It supports levels, formatting, and can be redirected to files without code changes.
