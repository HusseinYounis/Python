13.1 Series and DataFrames

A Series is a labeled 1D array. A DataFrame is a labeled 2D table — think of it as a dictionary of Series, or a spreadsheet/SQL table in memory.

import pandas as pd
import numpy as np

# Create DataFrame from dictionary
df = pd.DataFrame({
    "name": ["Alice", "Bob", "Charlie", "Diana"],
    "major": ["CE", "CS", "CE", "EE"],
    "gpa": [3.9, 3.5, 3.7, 3.8],
    "credits": [90, 75, 85, 95]
})

print(df)
print(df.shape)      # (4, 4)
print(df.dtypes)     # name: object, major: object, gpa: float64, credits: int64
print(df.describe()) # Statistical summary

13.2 Selection and Filtering

# Column access
print(df["name"])            # Series
print(df[["name", "gpa"]])   # DataFrame

# Row selection
print(df.loc[0])              # By label (index)
print(df.iloc[0:2])           # By position

# Boolean filtering
ce_students = df[df["major"] == "CE"]
honors = df[(df["gpa"] >= 3.7) & (df["credits"] >= 80)]

# Query syntax (SQL-like)
result = df.query("gpa > 3.6 and major == 'CE'")

13.3 Data Cleaning

# Handling missing data
df_dirty = pd.DataFrame({
    "sensor": ["A", "B", "C", "D"],
    "temp": [22.5, np.nan, 24.1, np.nan],
    "humidity": [45, 50, np.nan, 55]
})

print(df_dirty.isnull().sum())  # Count missing per column

# Fill missing values
df_filled = df_dirty.fillna({"temp": df_dirty["temp"].mean()})

# Drop rows with any missing
df_clean = df_dirty.dropna()

# String cleaning
df["name"] = df["name"].str.strip().str.lower()

13.4 Grouping and Aggregation

# Group by major
grouped = df.groupby("major")

# Aggregation
print(grouped["gpa"].mean())
print(grouped.agg({
    "gpa": ["mean", "max"],
    "credits": "sum"
}))

# Pivot table
pivot = df.pivot_table(
    values="gpa",
    index="major",
    aggfunc=["mean", "count"]
)

13.5 Merging and Joining

# Two DataFrames with a common key
courses = pd.DataFrame({
    "major": ["CE", "CS", "EE"],
    "dept": ["Computer Engineering", "Computer Science", "Electrical Engineering"]
})

# Merge (SQL JOIN)
merged = df.merge(courses, on="major", how="left")
print(merged)

# Concatenate vertically
df2 = pd.DataFrame({"name": ["Eve"], "major": ["CS"], "gpa": [3.6], "credits": [60]})
combined = pd.concat([df, df2], ignore_index=True)
