14.1 Basic Plots

Matplotlib is Python's foundational plotting library. It follows a MATLAB-like state-based interface (pyplot) alongside an object-oriented API.

import matplotlib.pyplot as plt
import numpy as np

# Line plot
x = np.linspace(0, 2 * np.pi, 100)
plt.figure(figsize=(8, 4))
plt.plot(x, np.sin(x), label="sin(x)", color="blue")
plt.plot(x, np.cos(x), label="cos(x)", color="red", linestyle="--")
plt.xlabel("x (radians)")
plt.ylabel("Amplitude")
plt.title("Trigonometric Functions")
plt.legend()
plt.grid(True, alpha=0.3)
plt.savefig("trig.png", dpi=150, bbox_inches="tight")
plt.show()

14.2 Common Plot Types

Scatter Plot

np.random.seed(42)
x = np.random.randn(200)
y = 0.5 * x + np.random.randn(200) * 0.3

plt.figure(figsize=(6, 6))
plt.scatter(x, y, alpha=0.6, c=y, cmap="coolwarm", edgecolors="black", linewidths=0.5)
plt.colorbar(label="y-value")
plt.xlabel("Feature 1")
plt.ylabel("Feature 2")
plt.title("Scatter Plot with Color Map")
plt.show()

Bar Chart

categories = ["CE", "CS", "EE", "ME"]
values = [45, 60, 35, 50]
colors = ["#3b82f6", "#60a5fa", "#93c5fd", "#bfdbfe"]

plt.figure(figsize=(8, 4))
bars = plt.bar(categories, values, color=colors, edgecolor="black")
plt.ylabel("Number of Students")
plt.title("Enrollment by Department")

# Add value labels on bars
for bar, val in zip(bars, values):
    plt.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 1,
             str(val), ha="center", fontweight="bold")
plt.show()

Histogram

data = np.random.normal(loc=70, scale=10, size=1000)

plt.figure(figsize=(8, 4))
plt.hist(data, bins=30, color="#3b82f6", edgecolor="white", alpha=0.8)
plt.axvline(data.mean(), color="red", linestyle="--", label=f"Mean: {data.mean():.1f}")
plt.xlabel("Score")
plt.ylabel("Frequency")
plt.title("Grade Distribution")
plt.legend()
plt.show()

14.3 Subplots

fig, axes = plt.subplots(2, 2, figsize=(10, 8))

# Top-left: line plot
axes[0, 0].plot(x, np.sin(x))
axes[0, 0].set_title("sin(x)")

# Top-right: scatter
axes[0, 1].scatter(np.random.randn(50), np.random.randn(50))
axes[0, 1].set_title("Random Scatter")

# Bottom-left: bar
axes[1, 0].bar(["A", "B", "C"], [10, 20, 15])
axes[1, 0].set_title("Bar Chart")

# Bottom-right: histogram
axes[1, 1].hist(np.random.randn(500), bins=20)
axes[1, 1].set_title("Histogram")

fig.suptitle("Four-Panel Figure", fontsize=14, fontweight="bold")
plt.tight_layout()
plt.show()

14.4 Customization and Styles

# Use a built-in style
plt.style.use("seaborn-v0_8-whitegrid")

# Custom color cycle
plt.rcParams["axes.prop_cycle"] = plt.cycler(
    color=["#1e3a8a", "#3b82f6", "#60a5fa", "#93c5fd"]
)

# Object-oriented approach for fine control
fig, ax = plt.subplots(figsize=(8, 4))
ax.plot(x, np.sin(x), linewidth=2)
ax.set_xlim(0, 2 * np.pi)
ax.set_ylim(-1.5, 1.5)
ax.set_xticks([0, np.pi/2, np.pi, 3*np.pi/2, 2*np.pi])
ax.set_xticklabels(["0", r"$\pi/2$", r"$\pi$", r"$3\pi/2$", r"$2\pi$"])
ax.spines["top"].set_visible(False)
ax.spines["right"].set_visible(False)
plt.show()

14.5 Seaborn for Statistical Plots

import seaborn as sns
import pandas as pd

# Sample data
df = pd.DataFrame({
    "major": ["CE"]*30 + ["CS"]*30 + ["EE"]*30,
    "gpa": np.concatenate([
        np.random.normal(3.5, 0.3, 30),
        np.random.normal(3.3, 0.4, 30),
        np.random.normal(3.6, 0.25, 30)
    ])
})

# Box plot
plt.figure(figsize=(8, 4))
sns.boxplot(x="major", y="gpa", data=df, palette="Blues")
plt.title("GPA Distribution by Major")
plt.show()

# Violin plot
plt.figure(figsize=(8, 4))
sns.violinplot(x="major", y="gpa", data=df, palette="Blues")
plt.title("GPA Violin Plot")
plt.show()

# Pair plot (scatter matrix)
iris = sns.load_dataset("iris")
sns.pairplot(iris, hue="species", palette="Blues")
plt.show()
